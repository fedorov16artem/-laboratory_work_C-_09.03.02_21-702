using namespace std;

int main () {
	system ("chcp 1251");
	
	int arraySize;
	
	cout << "Введите количество элементов массива: " << endl;
	cin >> arraySize;
	
	int array[arraySize];
	
	cout << "Одномерный массив из " << arraySize << " случайных чисел:" << endl;
	for (int i = 0; i < arraySize; i++) {
		array[i] = rand()%100;
		cout << "array[" << i << "] = " << array[i] << endl;
	}
	
	for (int i = 0; i < arraySize - 1; i++) {
		for (int j = 0; j < arraySize - i - 1; j++) {
			if (array[j] > array[j + 1]) {
				int temp = array[j];
				array[j] = array[j + 1];
				array[j + 1] = temp;
			}
		}
	}

	cout << "Простая сортировка \"Пузырьком\":" << endl;
	for (int i = 0; i < arraySize; i++) {
		cout << "array[" << i << "] = " << array[i] << endl;
	}
	
	system("pause");	
	return 0;
}
